\section{Proof of the Torsor Formulation}

The reductions and cyclic theories of the preceding sections leave one
explicit calculation. We state its three cyclic forms together and then prove
them in parallel. This is the only intermediate statement needed for the
proof of \Cref{theorem:torsors_after_blowup_is_tame}.

\begin{theorem}[The three cyclic cases]
\label{theorem:cyclic_cases_after_blowup}
\label{theorem:ramification_after_blowup_P1}
\label{cor:ramification_after_blowup_is_tame}
\label{theorem:ramification_after_blowup_is_tame_for_p_power}
Let $\cP\to U=C\setminus\mdls$ be a $G$-torsor with $G$ cyclic, and let
$\ndls=dP\subset\mdls$ be a single-point submodulus.
\begin{enumerate}
    \item If $G=\Z/e\Z$ with $(e,p)=1$, then $\cP^{[d]}$ is tamely
    ramified at $\eta_\ndls$.
    \item If $G=\Z/p\Z$ and the ramification of $\cP$ at $P$ is bounded
    by $d$, then $\cP^{[d]}$ is unramified at $\eta_\ndls$.
    \item If $G=\Z/p^r\Z$, $r\geq2$, and the ramification of $\cP$ at
    $P$ is bounded by $d$, then $\cP^{[d]}$ is unramified at
    $\eta_\ndls$.
\end{enumerate}
\end{theorem}

\begin{proof}
By \Cref{section:reductions_before_cyclic_theories}, we may work over an
algebraically closed field and regard $P$ as rational. Choose a uniformizer
$x$ at $P$, let
\[
    \rho:\Gal(k((x))^{\mathrm{sep}}/k((x)))\longrightarrow G
\]
classify the punctured-formal-disc restriction $\cP_x$, and put
$H:=\operatorname{im}(\rho)$. If $H$ is trivial, then $\cP_x$ and its
symmetric power are trivial, so the conclusion follows from
\Cref{lemma:formal_local_invariance}.

Assume $H\neq0$. The surjective character with image $H$ defines a connected
$H$-torsor $\cQ$ over $\Spec k((x))$, and
$\cP_x\cong\iota_*\cQ$ for the inclusion $\iota:H\hookrightarrow G$.
Whenever a ramification bound is assumed, it passes from $\cP_x$ to $\cQ$.
By
\Cref{lemma:realization_over_Gm}, the torsor $\cQ$ is realized by an explicit
connected $H$-torsor $\cQ'$ on $\bG_m$. Moreover,
\[
    (\iota_*\cQ')^{[d]}
       \cong\iota_*(\cQ'^{[d]})
\]
by \Cref{lemma:symmetric_power_induced_torsor}, and induction preserves the
ramification bound by \Cref{lemma:ramification_of_induced_torsor}. Therefore
\Cref{cor:reduction_to_P1} reduces every case to the connected explicit model
\[
    C=\bP^1_k,\qquad U=\bG_m,\qquad \ndls=d\cdot0.
\]
We rename $H$ as $G$ for the calculation.

Write $x_1,\dots,x_d$ for the coordinates on $\bG_m^d$ and
$e_1,\dots,e_d$ for their elementary symmetric polynomials. By
\Cref{lemma:formal_local_invariance}, the completed local ring and field at
the generic point $\eta=\eta_{d\cdot0}$ of the exceptional divisor are
\begin{equation}\label{eq:complete_field_at_generic_point}
    \widehat\Oo_\eta=\kappa[[e_d]],\qquad
    \widehat K_\eta=\kappa((e_d)),\qquad
    \kappa=k(e_1/e_d,\dots,e_{d-1}/e_d).
\end{equation}
The three calculations have the common shape
\[
\begin{array}{c|c|c|c}
 &\Z/e\Z&\Z/p\Z&\Z/p^r\Z\\ \hline
\text{normal form}&T^e=x^a&X^p-X=f&\wp\vec X=\vec f\\
\text{product coordinate}&\prod_iT_i&\sum_iX_i&\bigoplus_i\vec X_i\\
\text{new datum}&e_d^a&\sum_i f(x_i)&\bigoplus_i\vec f(x_i)\\
\text{exceptional behavior}&v(e_d^a)=a&\text{regular}&
    \text{componentwise regular}\\
\text{conclusion}&\text{tame}&\text{unramified}&\text{unramified}.
\end{array}
\]

\medskip
\noindent\textbf{Case (1): Kummer.}
The relevant group scheme is $\bG_m$. More precisely, the connected torsor is
the pullback of the finite \'etale homomorphism
\[
    [e]:\bG_m\longrightarrow\bG_m,\qquad T\longmapsto T^e,
\]
whose kernel is $\mu_e\cong G$, along the section $x\mapsto x^a$. Thus it has
equation
\[
    T^e=x^a,\qquad (a,e)=1,
\]
by \Cref{lemma:realization_over_Gm}(1). Put
\[
    R:=k[x,x^{-1}],\qquad S:=R[T]/(T^e-x^a).
\]
Then $\Spec S\to\Spec R=\bG_m$ is this $G$-torsor. After pulling it back
along the $d$ projections from $\bG_m^d$, the product torsor has coordinate
algebra $S^{\otimes_k d}$ over $R^{\otimes_k d}$. Contracting the product
action $G^d$ along $G^d\to G$ means taking invariants under
$\KG=\ker(G^d\to G)$. By
\Cref{prop:parallel_cyclic_product_torsor}, multiplication in $\bG_m$ gives
\[
    Y=\prod_{i=1}^dT_i,\qquad
    Y^e=\prod_{i=1}^dx_i^a=(x_1\cdots x_d)^a=e_d^a,
\]
and hence an isomorphism of $G$-torsor algebras
\[
    (S^{\otimes_k d})^{\KG}
       \cong
    R^{\otimes_k d}[Y]/(Y^e-e_d^a).
\]
The group $S_d$ permutes the tensor factors and fixes both $Y$ and $e_d^a$.
The algebra on the right is free over $R^{\otimes_k d}$ with the
$S_d$-invariant basis $1,Y,\dots,Y^{e-1}$, so taking $S_d$-invariants acts
only on its coefficients. Since
\[
    (R^{\otimes_k d})^{S_d}
       =k[x_1^{\pm1},\dots,x_d^{\pm1}]^{S_d}
       =k[e_1,\dots,e_d,e_d^{-1}],
\]
\Cref{cor:sym-power-quotient} identifies the descended torsor with
\[
    \cP^{[d]}\cong
    \Spec\frac{k[e_1,\dots,e_d,e_d^{-1}][Y]}{(Y^e-e_d^a)}.
\]
Its generic fiber over $\widehat K_\eta$ is therefore the restriction of
$\cP^{[d]}$ and is given by the same Kummer equation. The defining element
$e_d^a$ has valuation $a$; thus
\Cref{theorem:kummer_ramification}, together with $(e,p)=1$, shows that every
field factor is tamely ramified. This proves part~(1).

\medskip
\noindent\textbf{Case (2): Artin--Schreier.}
Here the relevant group scheme is $\bG_a$. The torsor is the pullback of the
finite \'etale homomorphism
\[
    \wp:\bG_a\longrightarrow\bG_a,\qquad X\longmapsto X^p-X,
\]
whose kernel is $\F_p\cong G$, along the section $x\mapsto f(x)$ on
$\bG_m$. By \Cref{lemma:realization_over_Gm}(2), its equation is
\[
    X^p-X=f(x),\qquad f\in x^{-1}k[x^{-1}],
\]
where either $f=0$, or $m:=\deg_{x^{-1}}f<d$ and $p\nmid m$. Put
\[
    R:=k[x,x^{-1}],\qquad
    S:=R[X]/(X^p-X-f(x)).
\]
Thus $\Spec S\to\Spec R=\bG_m$ is the realized $G$-torsor. After pullback
along the $d$ projections, $S^{\otimes_k d}$ is the coordinate algebra of
the product torsor over $R^{\otimes_k d}$. Contracting along $G^d\to G$
again amounts to taking $\KG$-invariants. This time the group law on
$\bG_a$ is addition, so \Cref{prop:parallel_cyclic_product_torsor} gives
\[
    Y=\sum_{i=1}^dX_i,\qquad
    Y^p-Y=\alpha,\qquad
    \alpha:=\sum_{i=1}^df(x_i),
\]
and hence
\[
    (S^{\otimes_k d})^{\KG}
       \cong
    R^{\otimes_k d}[Y]/(Y^p-Y-\alpha).
\]
The group $S_d$ fixes $Y$ and $\alpha$. The displayed algebra is free over
$R^{\otimes_k d}$ with the $S_d$-invariant basis
$1,Y,\dots,Y^{p-1}$, so taking $S_d$-invariants acts only on the
coefficients. Using $(R^{\otimes_k d})^{S_d}
=k[e_1,\dots,e_d,e_d^{-1}]$ and
\Cref{cor:sym-power-quotient}, we obtain
\[
    \cP^{[d]}\cong
    \Spec\frac{k[e_1,\dots,e_d,e_d^{-1}][Y]}{(Y^p-Y-\alpha)}.
\]
Put $y_i=x_i^{-1}$. If $f\neq0$, then $\alpha$ is a symmetric polynomial in
$y_1,\dots,y_d$ whose monomials have total degree at most $m<d$; if $f=0$,
then $\alpha=0$. Hence \Cref{lemma:symmetric_regularity} gives
\[
    \alpha\in\kappa\subset\widehat\Oo_\eta.
\]
Set $A:=\widehat\Oo_\eta$ and
\[
    B:=A[Y]/(Y^p-Y-\alpha).
\]
Because the defining polynomial is monic of degree $p$, the algebra $B$ is
finite free of rank $p$ over $A$. Its derivative with respect to $Y$ is
$-1$, a unit in $B$; hence $B/A$ is finite \'etale. The preceding descended
presentation identifies its generic fiber
\[
    B\otimes_A\widehat K_\eta
       =\widehat K_\eta[Y]/(Y^p-Y-\alpha)
\]
with the restriction of $\cP^{[d]}$ to $\widehat K_\eta$.

Since $A$ is a complete discrete valuation ring, the finite \'etale
$A$-algebra $B$ is a product of finite \'etale local $A$-algebras. Each
factor is a discrete valuation ring whose maximal ideal is generated by the
uniformizer of $A$, and whose residue field is finite separable over
$\kappa$. Consequently, every field factor of the generic fiber has
ramification index $1$ and separable residue extension, so it is unramified
in the sense of \Cref{definition:tamely_ramified_dvrs}. This proves part~(2).

\end{proof}

\begin{proof}[Proof of \Cref{theorem:torsors_after_blowup_is_tame}]
Fix a single-point submodulus $\ndls=dP$. By
\Cref{prop:reduction_to_cyclic_groups}, it is enough to consider cyclic
groups of order prime to $p$ and cyclic $p$-groups. The prime-to-$p$ case is
\Cref{theorem:cyclic_cases_after_blowup}(1). For $G=\Z/p^r\Z$, the
ramification bound on $\cP$ gives the unramified conclusion by part~(2) when
$r=1$ and by part~(3) when $r\geq2$. The reduction proposition reassembles
the cyclic factors and proves the theorem for every finite abelian group.
\end{proof}
